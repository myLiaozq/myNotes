
1、第一种
.element {
    
width: 600px; 
height: 400px;
    
position: absolute; 
left: 50%; 
top: 50%;
    
margin-top: -200px;    /* 高度的一半 */
    
margin-left: -300px;    /* 宽度的一半 */

}

2、第二种
.element {
    
width: 600px; 
height: 400px;
    
position: absolute; 
left: 50%; 
top: 50%;
    
transform: translate(-50%, -50%);    /* 50%为自身尺寸的一半 */

}

3、第三种
.element {
   
 width: 600px; 
height: 400px;
    
position: absolute; 
left: 0; 
top: 0; 
right: 0; 
bottom: 0;
    
margin: auto;    /* 有了这个就自动居中了 */

}
